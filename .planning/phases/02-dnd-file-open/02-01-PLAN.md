def _setup_file_drop(root, app):
    """windnd による D&D。未インストール時はスキップ。"""
    try:
        import windnd
        def on_drop(files):
            for f in files:
                path = f.decode('utf-8') if isinstance(f, bytes) else f
                if path.lower().endswith('.pdf'):
                    app._open_pdf_path(path)
                    break
        windnd.hook_dropfiles(root, func=on_drop)
    except ImportError:

      - `self.doc` が None の場合 → 直接 `self._open_pdf_path(pdf_paths[0])`
    - PDF が2件以上 → `MergeOrderDialog(self.root, list(pdf_paths), self._do_open_merged, lang=self.lang)`
    - `return event.action` を必ず返す
